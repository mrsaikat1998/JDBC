CREATE TABLE "SYSTEM"."CUSTOMER" 
   (	"CNO" NUMBER(10,0) NOT NULL ENABLE, 
	"CNAME" VARCHAR2(20 BYTE), 
	"CADD" VARCHAR2(20 BYTE), 
	"BILLAMOUNT" FLOAT(126), 
	 CONSTRAINT "CUSTOMER_PK" PRIMARY KEY ("CNO"))
	 
	 CREATE SEQUENCE  "SYSTEM"."CNO_SEQ"  MINVALUE 1 MAXVALUE 1000 INCREMENT BY 1 START WITH 1
